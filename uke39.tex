\problem{A.7.5}
\subproblem{a}
\begin{align*}
  \bar{z} &= 2-3i\\
  \bar{w} &= 1-6i\\
\end{align*}
\begin{align*}
  \bar{zw}&=\bar{z}\bar{w}\\
          &=(2-3i)(1-6i)\\
          &=2-12i-3i+18(i^2)\\
          &=2-15i-18\\
          &=-16-15i\\
\end{align*}
\subproblem{b}
\begin{align*}
  \bar{z+w} &= \bar{z}+\bar{w}\\
            &= (2-3i) + (1-6i)\\
            &= 2-3i +1-6i\\
            &=3-9i\\
\end{align*}
\subproblem{c}
\begin{align*}
  \bar{z/w} &= \bar{z}/\bar{w}\\
            &= (2-3i)/(1-6i)\\
            &= \frac{1\cdot 2 + (-6)(-3)}{1^2+(-6)^2} + i\frac{(-6)2-1(-3)}{1^2+(-6)^2}\\
            &= \frac{2+ 18}{1+36} + i\frac{-12+3}{1+36}\\
            &= \frac{20}{37} - i\frac{9}{37}\\
\end{align*}
